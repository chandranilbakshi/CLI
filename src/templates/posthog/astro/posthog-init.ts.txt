import posthog from 'posthog-js';

// PostHog client init for Astro. This module runs only in the browser bundle
// (Astro inlines `client:load` / `<script>` imports into client JS). We still
// guard with `typeof window` because the same file may be transitively
// imported during SSR — the guard prevents init from accidentally running on
// the server during static generation.
if (typeof window !== 'undefined') {
  const key = import.meta.env.PUBLIC_POSTHOG_KEY;
  if (key) {
    posthog.init(key, {
      api_host: import.meta.env.PUBLIC_POSTHOG_HOST || '{{HOST}}',
      capture_pageview: 'history_change',
      capture_pageleave: true,
    });
  }
}
