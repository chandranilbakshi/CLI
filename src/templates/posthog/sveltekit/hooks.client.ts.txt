import posthog from 'posthog-js';
import { browser } from '$app/environment';
import { PUBLIC_POSTHOG_KEY, PUBLIC_POSTHOG_HOST } from '$env/static/public';

// `hooks.client.ts` only runs in the browser, so we don't need an explicit
// `typeof window` guard. The `browser` import is included so future edits
// (e.g. moving init to a non-client hook) don't accidentally fire on the server.
if (browser && PUBLIC_POSTHOG_KEY) {
  posthog.init(PUBLIC_POSTHOG_KEY, {
    api_host: PUBLIC_POSTHOG_HOST || '{{HOST}}',
    capture_pageview: true,
    capture_pageleave: true,
  });
}

export const handleError = ({ error }: { error: unknown }) => {
  posthog.capture('$exception', { error: String(error) });
};
